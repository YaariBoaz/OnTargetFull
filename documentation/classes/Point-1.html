    isInSquare(sq, list) {
        let count = 0;
        for (const item in list) {
            if (this.inSquare(sq, list[item])) {
                count++;
            }
        }
        return count;
    }

    inSquare(sq, hit) {
        if ((hit.x > sq.A.X && hit.x <= sq.B.X) && (hit.y >= sq.A.Y && hit.y <= sq.C.Y)) {
            return true;
        }
        return false;
    }
